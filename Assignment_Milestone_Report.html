# create a worldcloud to see the frequency of words
freq_words <- sort(rowSums(matrx_dtm), decreasing = TRUE)
df_freq_words <- data.frame(Word = names(freq_words),freq=freq_words)
wordcloud(df_freq_words$Word,df_freq_words$freq,scale=c(4,0.8), min.freq = 5, max.words=100, random.order=FALSE,rot.per=0.35, colors=brewer.pal(8,"Accent"))

# plot the word frequencies
ggplot(df_freq_words, aes(x=reorder(Word,freq), y=freq, fill=freq)) + geom_bar(stat="identity") +
    coord_flip() + theme(axis.title.y = element_blank()) + labs(y="Frequency", title="Frequent Words")

# From both the worldcloud and the plot we can see that the words 'just', 'likes', 'one', 'can' & 'will' are the most frequent words
